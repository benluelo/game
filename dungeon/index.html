Crate dungeon[−][src]

Dungeon creation. Creates 2d cave-like dungeons using cellular automata (among other techniques).

Examples

use dungeon::{Dungeon, DungeonType};
use std::{
    convert::TryInto,
    num::NonZeroU16
};

let _ = Dungeon::new(
    150.try_into().unwrap(),
    100.try_into().unwrap(),
    NonZeroU16::new(10).unwrap(),
    DungeonType::Cave,
    false,
);
